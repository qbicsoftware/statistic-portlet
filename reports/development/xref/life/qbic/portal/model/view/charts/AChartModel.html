4   import life.qbic.portal.model.view.AModel;
5   import org.apache.logging.log4j.LogManager;

9   /**
10   * @author fhanssen
11   * This abstract class holds all fields and methods needed for a general Chart, mainly the configuration file. When
12   * adding a new chart type (such as Scatter Plot or so) a new chart model needs be added. A comprehensive
13   * guide on Vaadin charts can be found here: https://demo.vaadin.com/charts/
14   */
15  
16  
17  public abstract class AChartModel<T extends AbstractSeries> implements AModel<T> {
18  
19      private static final Logger logger = LogManager.getLogger(AChartModel.class);
20  
21      final Configuration configuration;
22      private final String title;
23      private final String subtitle;
24      private final String tabtitle;
25  
26  
27      AChartModel(Configuration configuration, String title, String subtitle, String tabTitle,
28                  Tooltip tooltip, Legend legend, AbstractPlotOptions plotOptions){
29  
30          this.title = title;
31          this.subtitle = subtitle;
32          this.tabtitle = tabTitle;
33  
34          this.configuration = configuration;
35          this.configuration.setTooltip(tooltip);
36          this.configuration.setLegend(legend);
37          this.configuration.setPlotOptions(plotOptions);
38  
39          logger.info("New chart model was successfully created");
40      }
41  
42      public Configuration getConfiguration() {
43          return configuration;
44      }
45  
46      @Override
47      abstract public void addData(T... item);
48  
49      @Override
50      public String getTitle(){
51          return title;
52      }
53  
54      @Override
55      public String getSubTitle(){
56          return subtitle;
57      }
58  
59      @Override
60      public String getTabTitle(){
61          return tabtitle;
62      }
63  
64      //TODO 2: If your chart TYPE does not exists yet, extend this class
65  }

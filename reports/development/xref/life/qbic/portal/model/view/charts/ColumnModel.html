5   import java.util.ArrayList;
6   import java.util.Arrays;
7   import java.util.List;

9   /**
10   * @author fhanssen
11   * Example: https://demo.vaadin.com/charts/#BasicColumn
12   */
13  public class ColumnModel extends AChartModel<ListSeries>{
14  
15      private final List<Series> series;
16      private final XAxis xAxis;
17      private final YAxis yAxis;
18  
19      public ColumnModel(Configuration configuration,String title, String subtitle, String tabTitle,
20                         Tooltip tooltip, Legend legend, AxisTitle xAxisTitle, AxisTitle yAxisTitle, PlotOptionsColumn options){
21  
22          super(configuration, title, subtitle, tabTitle, tooltip, legend,options);
23  
24          series = new ArrayList<>();
25          super.configuration.setSeries(series);
26          xAxis = super.configuration.getxAxis();
27          yAxis = super.configuration.getyAxis();
28          xAxis.setTitle(xAxisTitle);
29          yAxis.setTitle(yAxisTitle);
30          options.setAnimation(false);
31  
32      }
33  
34      public void addXCategories(String... cat){
35          xAxis.setCategories(cat);
36      }
37  
38      public void setYMin(int yMin){
39          yAxis.setMin(yMin);
40      }
41  
42      public void addData(ListSeries... listSeries){
43          series.addAll(Arrays.asList(listSeries));
44      }
45  
46      public void addData(DataSeries... dataSeries){
47          series.addAll(Arrays.asList(dataSeries));
48      }
49  
50      public void setXAxisType(AxisType type){
51          xAxis.setType(type);
52      }
53  
54      public void clearData(){
55          series.clear();
56      }
57  
58  }

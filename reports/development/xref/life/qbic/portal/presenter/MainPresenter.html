84          try {

86                  if (r.getName().equals("Administrator")) {

88                  }
89              }
90          } catch (Exception e) {
91              logger.error("Could not determine whether logged in user is admin.");

102     void addChildPresenter() throws DataNotFoundException{
103         clear();
104         int counter = 0;
105         for(ATabPresenter presenter : tabPresenterList){
106             try{
107                 presenter.setUp();
108                 presenter.addChart(new TabView((AView) presenter.getView(),
109                         (AModel/qbic/portal/model/view/AModel.html#AModel">AModel) presenter.getModel()), ((AModel) presenter.getModel()).getTabTitle());
110 
111             }catch(DataNotFoundException e){
112                 logger.error("Mainchart could not be set up.", e);
113                 counter++;//count up whenever exception is thrown
114                 //Exception was also thrown on last presenter
115                 if(counter > tabPresenterList.size() - 1){
116                     Styles.notification("Data not found.","Chart cannot be displayed", Styles.NotificationType.ERROR);
117                     throw new DataNotFoundException(e);
118                 }
119             }
120         }

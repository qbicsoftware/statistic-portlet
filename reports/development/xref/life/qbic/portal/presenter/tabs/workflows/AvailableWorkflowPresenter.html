22  public class AvailableWorkflowPresenter extends ATabPresenter<GridModel, GridView> {
23  
24      private final String chartName;
25      private ChartConfig chartConfig;
26  
27      AvailableWorkflowPresenter(MainPresenter mainPresenter, String chartName){
28          super(mainPresenter, new GridView(3,0,true, true));
29          this.chartName = chartName;
30      }
31  
32      @Override
33      protected void extractData() throws DataNotFoundException, NullPointerException {
34          chartConfig = super.getChartConfig(chartName);
35      }
36  
37      @Override
38      protected void addChartSettings() {
39          super.setModel(new GridModel(chartConfig.getSettings().getTitle(), chartConfig.getSettings().getSubtitle(), chartConfig.getSettings().getTabTitle()));
40          super.getView().getComponent().setStyleName("workflow");
41  
42          logger.info("Settings were added to " + this.getClass() + " with chart title: " + super.getModel().getTitle());
43      }
44  
45      @Override
46      protected void addChartData(){
47  
48          //This is necessary to get from Object to required String arrays
49          Object[] objectArray = chartConfig.getData().keySet().toArray(new Object[0]);
50          //noinspection SuspiciousToArrayCall
51          @SuppressWarnings("SuspiciousToArrayCall") String[] keySet = Arrays.asList(objectArray).toArray(new String[objectArray.length]);
52  
53  
54          //Actually adding of data
55          for (String aKeySet : keySet) {
56              for (int i = 0; i < chartConfig.getData().get(aKeySet).size(); i++) {
57                  String[] title = ((String) chartConfig.getSettings().getxCategories().get(i)).split("/");
58                  super.getModel().addData(new GitHubLabels(title[title.length - 2].concat("/").concat(title[title.length - 1]),
59                          (String) chartConfig.getSettings().getyCategories().get(i),
60                          (int)(double) chartConfig.getData().get(aKeySet).get(i)));
61              }
62          }
63  
64          logger.info("Data was added to a chart of " + this.getClass() + " with chart title: " + chartConfig.getSettings().getTitle());
65  
66  
67      }
68  
69      @Override
70      public void addChart(TabView tabView, String title) {
71  
72          super.getView().setRows(super.getModel().getData().size()/2 + 1);
73          for (Object labels : super.getModel().getData()) {
74              setGridItemLayout((GitHubLabels) labels);
75          }
76  
77          logger.info("Tab was added in " + this.getClass() + " for " +  super.getModel().getTitle());
78  
79      }
80  
81      private void setGridItemLayout(GitHubLabels labels) {
82  
83          Panel panel = new Panel(labels.getTitle());
84          panel.setHeight(100.0f, Sizeable.Unit.PERCENTAGE);
85          panel.setStyleName(ValoTheme.PANEL_SCROLL_INDICATOR);
86  
87          Label star = new Label(FontAwesome.STAR_O.getHtml() + " " + labels.getCount(), ContentMode.HTML);
88  
89          Link link = new Link("",
90                  new ExternalResource(labels.getUrl()));
91          link.setIcon(FontAwesome.valueOf("GITHUB_SQUARE"));
92  
93          HorizontalLayout horizontalLayout = new HorizontalLayout(link, star);
94          horizontalLayout.setSpacing(true);
95          horizontalLayout.setComponentAlignment(star, Alignment.TOP_RIGHT);
96          horizontalLayout.setComponentAlignment(link, Alignment.TOP_RIGHT);
97  
98          Label label = new Label(labels.getDescription(), ContentMode.HTML);
99          VerticalLayout verticalLayout = new VerticalLayout();
100         verticalLayout.setWidth(100, Sizeable.Unit.PERCENTAGE);
101         verticalLayout.setSpacing(false);
102         verticalLayout.addComponent(horizontalLayout);
103         verticalLayout.addComponent(label);
104         verticalLayout.setMargin(true);
105 
106         panel.setContent(verticalLayout);
107         super.getView().addGridComponents(panel);
108     }
109 
110     String getChartName() {
111         return chartName;
112     }
113 
114     @Override
115     protected void addChartListener(){}
116 }

116     @Override
117     protected void addChartData() {
118 
119         //This is necessary to get from Object to required String arrays
120         Object[] objectArray = kingdomConfig.getData().keySet().toArray(new Object[0]);
121         @SuppressWarnings("SuspiciousToArrayCall") String[] keySet = Arrays.asList(objectArray).toArray(new String[objectArray.length]);
122 
123         List<DataSorter> dataSorters = new ArrayList<>();
124 
125         //Retrieve and Sort data
126         for (String aKeySet : keySet) {
127             for (int i = 0; i < kingdomConfig.getData().get(aKeySet).size(); i++) {
128                 String label = LabelFormatter.generateCamelCase((String) kingdomConfig.getSettings().getxCategories().get(i))
129                         .concat(" [")
130                         .concat(String.valueOf(kingdomConfig.getSettings().getyCategories().get(i)))
131                         .concat("%]");
132                 dataSorters.add(new DataSorter(label,
133                         (int) kingdomConfig.getData().get(aKeySet).get(i)));
134 
135             }
136         }
137 
138         Collections.sort(dataSorters);
139 
140         //Add data
141 
142 
143         dataSorters.forEach(d -> {
144             DataSeriesItem dataSeriesItem = new DataSeriesItem(d.getName(), d.getCount());
145             //Either 'Kingdoms' or 'Other Kingdom' is clickable and thus is selected on hover
146             if(!isKingdom(d.getName())) {
147                 dataSeriesItem.setSelected(true);
148             }
149             this.getModel().addData(dataSeriesItem);
150         });
151 
152         logger.info("Data was added to a chart of  " + this.getClass() + "  with chart title: " + this.getView().getConfiguration().getTitle().getText());
153 
154     }

156     private boolean isKingdom(String name){
157         //returns true if name is of format <kingdom> [x%] or Other <kingdom> [x%]
158         //else returns false
159         return Kingdoms.getList().contains(name.split(" ")[0]) || Kingdoms.getList().contains(name.split(" ")[1]);
160     }

173     @Override
174     protected void addChartListener() {
175         ((Chart) getView().getComponent()).addPointClickListener((PointClickListener) event -> {
176 
177             logger.info("Chart of " + this.getClass() + " with chart title: " +
178                     this.getView().getConfiguration().getTitle().getText() +
179                     " was clicked at " + this.getModel().getDataName(event));
180 
181             //In case it is Other Bacteria etc.
182             if (Kingdoms.getList().contains(this.getModel().getDataName(event).split(" ")[1])) {
183                 ATabPresenter p =
184                         new GenusSpeciesCountPresenter(super.getMainPresenter(), this.getModel().getDataName(event).split(" ")[1]);
185                 addSubchart(p);
186             }
187 
188             //In case it is not Other, but plain 'Viruses' etc.
189             if (Kingdoms.getList().contains(this.getModel().getDataName(event).split(" ")[0])) {
190                 ATabPresenter p =
191                         new GenusSpeciesCountPresenter(super.getMainPresenter(), this.getModel().getDataName(event).split(" ")[0]);
192 
193                 addSubchart(p);
194             }
195 
196         });
197 
198         ((Chart) getView().getComponent()).addLegendItemClickListener(legendItemClickEvent -> {
199             //do nothing, overwrites normal function: hide/show clicked legend item in chart
200         });

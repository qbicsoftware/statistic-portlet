29          super(mainPresenter, new PieView());

32      @Override
33      protected void extractData() throws DataNotFoundException, NullPointerException {
34          workflowUsageConfig = super.getChartConfig(ChartNames.Workflow_Execution_Counts.toString());

37      @Override
38      protected void addChartSettings() {
39  
40          PlotOptionsPie plot = new PlotOptionsPie();
41  
42          plot.setDataLabels(new DataLabels(true));
43          plot.getDataLabels().setFormatter("'<span style=\"color:CornflowerBlue;text-decoration:underline\">' + this.point.name + '</span>'");
44          plot.setShowInLegend(true);
45  
46          Hover hover = new Hover();
47          hover.setEnabled(true);
48          plot.getStates().setHover(hover);
49  
50          Tooltip tooltip = new Tooltip();
51          tooltip.setFormatter("this.point.name + ': <b>'+ this.y + '</b> times executed <br> Click to show available workflows'");
52  
53          Legend legend = new Legend();
54          legend.setLabelFormatter("function() {" +
55                  "var text = this.name.split('[')[0];" +
56                  "text = text.substring(0, text.length - 1);" +
57                  "return text + ': ' + this.y + ' Runs' "+
58                  "}");
59          legend.setLayout(LayoutDirection.VERTICAL);
60          legend.setVerticalAlign(VerticalAlign.BOTTOM);
61          legend.setAlign(HorizontalAlign.RIGHT);
62  
63          super.setModel(new PieChartModel(super.getView().getConfiguration(), workflowUsageConfig.getSettings().getTitle(),
64                  workflowUsageConfig.getSettings().getSubtitle(),workflowUsageConfig.getSettings().getTabTitle(), tooltip, legend, plot));
65          logger.info("Settings were added to " + this.getClass() + " with chart title: " + super.getView().getConfiguration().getTitle().getText());
66  
67  
68      }
69  
70      @Override
71      protected void addChartData() {
72  
73          //This is necessary to get from Object to required String arrays
74          Object[] objectArray = workflowUsageConfig.getData().keySet().toArray(new Object[0]);
75          @SuppressWarnings("SuspiciousToArrayCall") String[] keySet = Arrays.asList(objectArray).toArray(new String[objectArray.length]);
76  
77          List<DataSorter> dataSorters = new ArrayList<>();
78  
79          //Actually adding of data
80          for (String aKeySet : keySet) {
81              for (int i = 0; i < workflowUsageConfig.getData().get(aKeySet).size(); i++) {
82                  String label = (String) workflowUsageConfig.getSettings().getxCategories().get(i);
83  
84                  if(CommonAbbr.getList().contains(label)){
85                      label = CommonAbbr.valueOf(label).toString();
86                  }else if(Translator.getList().contains(label)){
87                      label = Translator.valueOf(label).getTranslation();
88                  }else{
89                      label = LabelFormatter.generateCamelCase(label);
90                  }
91  
92                   label = label
93                          .concat(" [")
94                          .concat(String.valueOf(workflowUsageConfig.getSettings().getyCategories().get(i)))
95                          .concat("%]");
96                  dataSorters.add(new DataSorter(label,
97                          (int) workflowUsageConfig.getData().get(aKeySet).get(i)));
98              }
99          }
100 
101         Collections.sort(dataSorters);
102 
103         //Add data
104         dataSorters.forEach(d -> this.getModel().addData(new DataSeriesItem(d.getName(), d.getCount())));
105 
106         logger.info("Data was added to a chart of " + this.getClass() + " with chart title: " + workflowUsageConfig.getSettings().getTitle());
107 
108     }

122     @Override
123     protected void addChartListener() {
124         ((Chart) super.getView().getComponent()).addPointClickListener((PointClickListener) event -> {
125             logger.info("Chart of " + this.getClass() + " with chart title: " + super.getView().getConfiguration().getTitle().getText() + " was clicked at " + super.getModel().getDataName(event));
126 
127             String title =  ChartNames.Available_Workflows_.toString().replace("_", " ").trim();
128             String chartName = super.getModel().getDataName(event).replaceFirst("\\[[0-9]+.[0-9]+\\%\\]", "").trim(); //remove percentage from name: NGS [3.6%] -> NGS
129             try {
130                 ATabPresenter wfPresenter = new WorkflowTypePresenter(getMainPresenter(), chartName, title.concat(" for ").concat(chartName).concat(" Data Analysis"), "", "");
131                 addSubchart(wfPresenter);
132             }catch(Exception e){
133                 logger.error(chartName.concat(" workflows could not be displayed."), e.getMessage());
134             }
135         });
136 
137         ((Chart) getView().getComponent()).addLegendItemClickListener(legendItemClickEvent -> {
138             //do nothing, overwrites normal function: hide/show clicked legend item in chart
139         });
140     }

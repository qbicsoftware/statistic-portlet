17   * @author fhanssen
18   */
19  public class TabView extends VerticalLayout {
20  
21      private static final Logger logger = LogManager.getLogger(TabView.class);
22  
23      private final Button returnButton = new Button("Back", FontAwesome.ANGLE_DOUBLE_LEFT);
24      private final AView mainView;
25      private final AModel mainModel;
26      private final Label title;
27      private final Label subtitle;
28  
29      private final Panel graphPanel = new Panel();
30  
31      public TabView(AView view, AModel model){
32          this.mainView = view;
33          this.mainModel = model;
34  
35          this.title = new Label("", ContentMode.HTML);
36          title.setStyleName(ValoTheme.LABEL_NO_MARGIN);
37          title.setStyleName(ValoTheme.LABEL_BOLD);
38          title.addStyleName(ValoTheme.LABEL_COLORED);
39          title.addStyleName(ValoTheme.LABEL_HUGE);
40          title.addStyleName(ValoTheme.LABEL_H1);
41          title.setWidth(null);
42  
43          this.subtitle = new Label("", ContentMode.HTML);
44          subtitle.setStyleName(ValoTheme.LABEL_NO_MARGIN);
45          subtitle.setStyleName(ValoTheme.LABEL_BOLD);
46          subtitle.addStyleName(ValoTheme.LABEL_COLORED);
47          subtitle.addStyleName(ValoTheme.LABEL_LARGE);
48          subtitle.addStyleName(ValoTheme.LABEL_H1);
49          subtitle.setWidth(null);
50  
51          setMargin(false);
52          setSpacing(false);
53  
54          addComponent(title);
55          addComponent(subtitle);
56          setComponentAlignment(title, Alignment.MIDDLE_CENTER);
57          setComponentAlignment(subtitle, Alignment.MIDDLE_CENTER);
58  
59          returnButton.addStyleName(ValoTheme.BUTTON_BORDERLESS_COLORED);
60          returnButton.setWidth(10, Unit.PERCENTAGE);
61          returnButton.setHeight(10, Unit.PERCENTAGE);
62  
63          graphPanel.setWidth(100, Unit.PERCENTAGE);
64          //this.mainView.getComponent().setHeight(String.valueOf(this.mainView.getComponent().getHeight() + 50.0));
65  
66      }
67  
68      public Button getReturnButton() {
69          return returnButton;
70      }
71  
72      public void addMainComponent(){
73  
74          removeComponents();
75  
76          if(mainView instanceof AChartView) {
77              ((AChartView) mainView).draw((AChartModel) mainModel);
78          }
79          graphPanel.setContent(mainView.getComponent());
80  
81          setTitle(mainModel.getTitle());
82          setSubtitle(mainModel.getSubTitle());
83  
84          addComponents(this.title, this.subtitle, graphPanel);
85  
86          setComponentAlignment(this.title, Alignment.MIDDLE_CENTER);
87          setComponentAlignment(this.subtitle, Alignment.MIDDLE_CENTER);
88  
89          logger.info("Main component was added.");
90  
91      }
92  
93      private void removeComponents() {
94          removeAllComponents();
95          setTitle("");
96          setSubtitle("");
97          logger.info("All components were removed from tab.");
98      }
99  
100     public void addSubComponent(AModel model, AView view){
101 
102         removeComponents();
103 
104         if(view instanceof AChartView) {
105             ((AChartView)view).draw((AChartModel) model);
106             //FIXME: Something is off with vaadin charts size (multiple open issues on GitHub).
107             //FIXME: Using this work-around for now. The labels are not accounted for in the height measure,
108             //FIXmE: so adding a padding manually.
109             view.getComponent().setHeight(Double.toString(view.getComponent().getHeight() + 50.0f));
110         }
111         graphPanel.setContent(view.getComponent());
112 
113         setTitle(model.getTitle());
114         setSubtitle(model.getSubTitle());
115 
116         addComponents(returnButton,this.title, this.subtitle, graphPanel);
117 
118         setComponentAlignment(this.title, Alignment.TOP_CENTER);
119         setComponentAlignment(this.subtitle, Alignment.TOP_CENTER);
120         setComponentAlignment(returnButton, Alignment.TOP_LEFT);
121 
122         logger.info("Sub-component and return button was added to tab.");
123 
124     }
125 
126     private void setTitle(String title) {
127         this.title.setValue(title);
128     }
129 
130     private void setSubtitle(String subtitle){
131         this.subtitle.setValue(subtitle);
132     }
133 
134 }

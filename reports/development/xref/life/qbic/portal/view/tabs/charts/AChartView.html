10  /**
11   * @author fhanssen
12   * This abstract class holds the chart. When
13   * adding a new chart type (such as Scatter Plot or so) a new chart view needs be added. A comprehensive
14   * guide on Vaadin charts can be found here: https://demo.vaadin.com/charts/
15   */
16  public abstract class AChartView extends AView {
17  
18      private final Chart chart;
19  
20      AChartView(ChartType type){
21          super();
22          this.chart = new Chart(type);
23      }
24  
25      public Configuration getConfiguration(){
26          return this.chart.getConfiguration();
27      }
28  
29      public void draw(AChartModel model){
30          chart.drawChart(model.getConfiguration());
31      }
32  
33      @Override
34      public Component getComponent() {
35          return chart;
36      }
37  
38  
39      //TODO 3: If your Chart TYPE does not exist extend this class
